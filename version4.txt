#include <iostream>
#include <fstream>
#include <conio.h>
using namespace std;
bool gameOver;
const int plotis = 20;
const int aukstis = 20;
int x, y, maistX, maistY, rez;
enum eDirection { STOP = 0, LEFT, RIGHT, UP, DOWN};
eDirection dir;

void Setupas() 
{
	gameOver = false;
	dir = STOP;
	x = plotis / 2;
	y = aukstis / 2;
	maistX = rand() % plotis;
	maistY = rand() % aukstis;
	rez = 0;
}
void render()
{
	system("cls");
	for (int i = 0; i < plotis+2; i++)
		cout << "$";
	cout << endl;

	for (int i = 0; i < aukstis; i++)
	{
		for (int j = 0; j < plotis; j++)
		{
			if (j == 0)
				cout << "$";
			if (i == y && j == x)
				cout << "O";
			else if (i == maistY && j == maistX)
				cout << "F";
			else
				cout << " ";

			if (j == plotis - 1)
				cout << "$";
		}
		cout << endl;
	}
	for (int i = 0; i < plotis + 2; i++)
		cout << "$";
	cout << endl;
}
void Inputas()
{
	if (_kbhit())
	{
		switch (_getch())
		{
		case 'a':
			dir = LEFT;
			break;
		case 'd':
			dir = RIGHT;
			break;
		case 'w':
			dir = UP;
			break;
		case 's':
			dir = DOWN;
			break;
		case 'x':
			gameOver = true;
			break;
		}
	}
}
void LogikaZ() 
{
	switch (dir)
	{
	case LEFT:
		x--;
		break;
	case RIGHT:
		x++;
		break;
	case UP:
		y--;
		break;
	case DOWN:
		y++;
		break;
	default:
		break;
	}
}
	int main()
{
	Setupas();
	while (!gameOver)
	{
		render();
		Inputas();
	}
}
